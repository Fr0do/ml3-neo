% normalization.tex — BN / LN / IN / GN / RMSNorm side-by-side.
% Используется в lectures/03-mlp-norm и lectures/04-cnn (как припоминание).
% Идея визуализации: тензор [N, C, S] в изометрической проекции;
% подсветкой выделена область, по которой считаются μ и σ².
%
% Скомпилировать: python figures/render.py figures/tikz/normalization.tex
% Подключить в .qmd: ![](../figures/tikz/_build/normalization.svg)
%
% Тип нормализации передаётся числом:
%   1=BN  2=LN  3=IN  4=GN  5=RMSNorm

\documentclass[tikz,border=8pt]{standalone}
\input{_preamble.tex}

\begin{document}

% ---- Изометрические единичные векторы (в mm) ---------------------------------
% Ось C → вправо-вниз:  (ex, -0.5·ex)
% Ось S → вверх:        (0, ey)
% Ось N → влево-вниз:  (-ex, -0.5·ex)
\pgfmathsetmacro\ex{3.6}   % горизонтальный масштаб
\pgfmathsetmacro\ey{3.6}   % вертикальный масштаб
\pgfmathsetmacro\ez{2.0}   % глубина (N-ось)

% Изо-смещение на 1 шаг по каждой оси (в mm):
%   dC = (ex, -0.5*ex), dS = (0, ey), dN = (-ez, -0.5*ez)
\pgfmathsetmacro\dCx{\ex}
\pgfmathsetmacro\dCy{-0.5*\ex}
\pgfmathsetmacro\dSx{0}
\pgfmathsetmacro\dSy{\ey}
\pgfmathsetmacro\dNx{-\ez}
\pgfmathsetmacro\dNy{-0.5*\ez}

% --- Макрос: один куб в позиции (ci, si, ni), 0-based ----------------------
% #1=fill color, #2=ci, #3=si, #4=ni
\newcommand{\isocube}[4]{
  % Корень куба (нижний-левый-ближний угол)
  \pgfmathsetmacro\Ox{(#2)*\dCx + (#3)*\dSx + (#4)*\dNx}
  \pgfmathsetmacro\Oy{(#2)*\dCy + (#3)*\dSy + (#4)*\dNy}
  % Четыре вершины фронтальной грани (C-S плоскость, n=const):
  % P00 = (Ox, Oy)
  % P10 = P00 + dC
  % P11 = P00 + dC + dS
  % P01 = P00 + dS
  \pgfmathsetmacro\Axx{\Ox}                   \pgfmathsetmacro\Axy{\Oy}
  \pgfmathsetmacro\Bxx{\Ox+\dCx}              \pgfmathsetmacro\Bxy{\Oy+\dCy}
  \pgfmathsetmacro\Cxx{\Ox+\dCx}              \pgfmathsetmacro\Cxy{\Oy+\dCy+\dSy}
  \pgfmathsetmacro\Dxx{\Ox}                   \pgfmathsetmacro\Dxy{\Oy+\dSy}
  % Верхняя грань (C-N, s=top):
  \pgfmathsetmacro\Exx{\Ox+\dNx}              \pgfmathsetmacro\Exy{\Oy+\dSy+\dNy}
  \pgfmathsetmacro\Fxx{\Ox+\dCx+\dNx}        \pgfmathsetmacro\Fxy{\Oy+\dCy+\dSy+\dNy}
  % Боковая правая грань (N-S, c=right):
  \pgfmathsetmacro\Gxx{\Ox+\dCx+\dNx}        \pgfmathsetmacro\Gxy{\Oy+\dCy+\dNy}
  % Рисуем три видимые грани от тёмного к светлому
  % Правая грань (N-S): самая тёмная
  \fill[#1!50, draw=ml3ink, line width=0.2pt]
    (\Bxx mm, \Bxy mm) -- (\Gxx mm, \Gxy mm)
    -- (\Fxx mm, \Fxy mm) -- (\Cxx mm, \Cxy mm) -- cycle;
  % Верхняя грань (C-N): средняя
  \fill[#1!80, draw=ml3ink, line width=0.2pt]
    (\Dxx mm, \Dxy mm) -- (\Cxx mm, \Cxy mm)
    -- (\Fxx mm, \Fxy mm) -- (\Exx mm, \Exy mm) -- cycle;
  % Фронтальная грань (C-S): самая светлая
  \fill[#1, draw=ml3ink, line width=0.2pt]
    (\Axx mm, \Axy mm) -- (\Bxx mm, \Bxy mm)
    -- (\Cxx mm, \Cxy mm) -- (\Dxx mm, \Dxy mm) -- cycle;
}

% --- Макрос панели ----------------------------------------------------------
% #1 = TikZ-координата смещения (напр. {(0,0)})
% #2 = заголовок
% #3 = подпись формулы
% #4 = тип: 1=BN 2=LN 3=IN 4=GN 5=RMS
\newcommand{\normpanel}[4]{
  \begin{scope}[shift={#1}]
    % Painter's algorithm: рисуем от n=3 (дальний) до n=0 (ближний)
    \foreach \ni in {3,2,1,0} {
      \foreach \ci in {0,1,2,3,4} {
        \foreach \si in {0,1,2,3} {
          % Вычисляем флаг подсветки числом 0 или 1
          \pgfmathtruncatemacro\hlBN{(\ci==1) ? 1 : 0}
          \pgfmathtruncatemacro\hlLN{(\ni==2) ? 1 : 0}
          \pgfmathtruncatemacro\hlIN{(\ni==2 && \ci==2) ? 1 : 0}
          \pgfmathtruncatemacro\hlGN{(\ni==2 && \ci>=1 && \ci<=3) ? 1 : 0}
          \pgfmathtruncatemacro\hlRM{(\ni==2) ? 1 : 0}

          \pgfmathtruncatemacro\useHL{
            (#4==1 && \hlBN) || (#4==2 && \hlLN) ||
            (#4==3 && \hlIN) || (#4==4 && \hlGN) ||
            (#4==5 && \hlRM)
          }

          \ifnum\useHL=1
            \isocube{ml3accent}{\ci}{\si}{\ni}
          \else
            \isocube{ml3soft}{\ci}{\si}{\ni}
          \fi
        }
      }
    }

    % Метки осей
    \pgfmathsetmacro\axisCx{2.5*\dCx - 0.5}
    \pgfmathsetmacro\axisCy{2.5*\dCy - 4}
    \node[font=\scriptsize\color{ml3muted}] at (\axisCx mm, \axisCy mm) {$C$};

    \pgfmathsetmacro\axisSx{-3}
    \pgfmathsetmacro\axisSy{2.0*\dSy}
    \node[font=\scriptsize\color{ml3muted}, anchor=east] at (\axisSx mm, \axisSy mm) {$S$};

    \pgfmathsetmacro\axisNx{3.5*\dNx + 4.5*\dCx}
    \pgfmathsetmacro\axisNy{3.5*\dNy + 4.5*\dCy + 4.5*\dSy}
    \node[font=\scriptsize\color{ml3muted}, anchor=south] at (\axisNx mm, \axisNy mm) {$N$};

    % Заголовок и подпись
    \pgfmathsetmacro\titleX{2.5*\dCx}
    \pgfmathsetmacro\titleY{4.5*\dSy + 3.5*\dNy + 7}
    \node[font=\small\bfseries\color{ml3ink}, anchor=south]
      at (\titleX mm, \titleY mm) {#2};
    \node[font=\scriptsize\color{ml3muted}, anchor=north]
      at (\titleX mm, {\titleY mm - 4mm}) {#3};
  \end{scope}
}

% --- Шаг между панелями (ширина одной панели + отступ) ----------------------
% Ширина ≈ C*dCx + N*|dNx| + gap
\pgfmathsetmacro\panelstep{5*\dCx + 4*\ez + 8}

\begin{tikzpicture}

\normpanel{(0mm,0mm)}{BatchNorm}{по $N{\times}S$}{1}
\normpanel{(\panelstep mm,0mm)}{LayerNorm}{по $C{\times}S$}{2}
\normpanel{(2*\panelstep mm,0mm)}{InstanceNorm}{по $S$}{3}
\normpanel{(3*\panelstep mm,0mm)}{GroupNorm}{по $C_{g}{\times}S$}{4}
\normpanel{(4*\panelstep mm,0mm)}{RMSNorm}{по $C{\times}S$, без $\mu$}{5}

% Общая формула снизу
\pgfmathsetmacro\formulaX{2*\panelstep}
\node[anchor=north, font=\footnotesize, text width=70mm, align=center,
      inner sep=3mm, draw=ml3muted, rounded corners=3pt, text=ml3ink]
  at (\formulaX mm, -8mm)
{
  \textbf{Общая форма:}\;
  $\hat x = \dfrac{x - \mu}{\sqrt{\sigma^2 + \varepsilon}}$,\;
  $y = \gamma \hat x + \beta$\\[3pt]
  {\color{ml3accent}$\blacksquare$} — область усреднения $(\mu,\sigma^2)$
};

\end{tikzpicture}
